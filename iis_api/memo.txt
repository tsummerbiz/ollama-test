dotnet new web -n JwtAuthApi
cd JwtAuthApi

# Windows認証ライブラリ (8系を指定)
dotnet add package Microsoft.AspNetCore.Authentication.Negotiate --version 8.*

# JWTライブラリ (8系を指定)
dotnet add package System.IdentityModel.Tokens.Jwt --version 8.*

dotnet add package Serilog.AspNetCore --version 8.*
dotnet add package Serilog.Sinks.File --version 8.*



dotnet build

dotnet publish -c Release -r win-x64 --self-contained true -p:PublishSingleFile=true -p:DebugType=embedded
